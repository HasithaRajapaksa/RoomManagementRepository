2. Sql query for mutiple reservations 

SELECT user.Username, count(reservation.rid)
FROM user
LEFT JOIN reservation
ON user.id =reservation.userid 
where reservation.datein between '2022-01-14'AND '2022-02-13'
group by user.Username;


3. Reservation Table


CREATE TABLE Reservation(
  rid int,
  roomid int,
  userid int,
  date varchar(255),
  hours int,
  PRIMARY KEY (rid),
  FOREIGN KEY (roomid) REFERENCES rooms(id),
  FOREIGN KEY (userid) REFERENCES user(id)
                        
);
